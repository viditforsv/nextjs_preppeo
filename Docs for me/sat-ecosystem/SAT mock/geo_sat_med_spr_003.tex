\section*{SAT Geometry - Practice Set 9 (Medium Numeric Entry)}

% Q1: Based on Similar Triangle Scale Factor logic
\question
The table below gives the perimeters of two similar triangles, $\triangle ABC$ and $\triangle DEF$, where side $\overline{AB}$ corresponds to side $\overline{DE}$. 
\begin{center}
\begin{tabular}{|c|c|}
\hline
\textbf{Triangle} & \textbf{Perimeter (cm)} \\ \hline
$\triangle ABC$ & 40 \\ \hline
$\triangle DEF$ & 280 \\ \hline
\end{tabular}
\end{center}
If the length of side $\overline{AB}$ is 12 cm, what is the length of side $\overline{DE}$?

\begin{solution}
For similar triangles, the ratio of any two corresponding linear measurements (like sides or perimeters) is equal to the scale factor $k$.
\vspace{5pt}
\begin{center}
Scale Factor $k = \dfrac{\text{Perimeter of } \triangle DEF}{\text{Perimeter of } \triangle ABC}$
\vspace{5pt}
$k = \dfrac{280}{40} = 7$
\vspace{5pt}
Since the scale factor is 7, the length of side $\overline{DE}$ is 7 times the length of its corresponding side $\overline{AB}$.
\vspace{5pt}
$DE = 7 \times 12 = 84$
\end{center}
\vspace{5pt}
The length of side $\overline{DE}$ is 84 cm.
\vspace{5pt}
Answer: \colorbox{yellow!100}{84}
\end{solution}

% Q2: Based on Cylinder Volume logic
\question
A cylindrical storage tank has a diameter of 12 feet and a height of 10 feet. A second cylindrical tank has the same height, but its radius is $25\%$ larger than the radius of the first tank. What is the volume, in cubic feet, of the second tank? (Express your answer in terms of $\pi$).
\begin{center}
\begin{tikzpicture}[scale=0.8]
    % Tank 1
    \draw[thick] (0,0) ellipse (1 and 0.3);
    \draw[thick] (-1,0) -- (-1,-2);
    \draw[thick] (1,0) -- (1,-2);
    \draw[thick] (-1,-2) arc (180:360:1 and 0.3);
    \node at (0,0.5) {\tiny $d=12$};
    \node[left] at (-1,-1) {\tiny $h=10$};
    
    % Tank 2
    \draw[thick] (4,0) ellipse (1.25 and 0.375);
    \draw[thick] (2.75,0) -- (2.75,-2);
    \draw[thick] (5.25,0) -- (5.25,-2);
    \draw[thick] (2.75,-2) arc (180:360:1.25 and 0.375);
    \node at (4,0.6) {\tiny Radius +25\%};
\end{tikzpicture}
\end{center}

\begin{solution}
First, determine the radius of the first tank ($r_1$). The diameter is 12, so the radius is 6 feet.
\vspace{5pt}
\begin{center}
Next, calculate the radius of the second tank ($r_2$):
\vspace{5pt}
$r_2 = 6 + (0.25 \times 6) = 6 + 1.5 = 7.5 \text{ feet}$.
\vspace{5pt}
Now, use the volume formula for a cylinder, $V = \pi r^2 h$, for the second tank:
\vspace{5pt}
$V = \pi (7.5)^2 (10)$
\vspace{5pt}
$V = \pi (56.25)(10) = 562.5\pi$
\end{center}
\vspace{5pt}
The volume of the second tank is $562.5\pi$ cubic feet.
\vspace{5pt}
Answer: \colorbox{yellow!100}{562.5}
\end{solution}

% Q3: Based on 30-60-90 Special Right Triangle logic
\question
In the figure below, $\triangle PQR$ is a right triangle with a right angle at $Q$. If the measure of $\angle R$ is $30^\circ$ and the length of the hypotenuse $\overline{PR}$ is 26, what is the length of side $\overline{PQ}$?
\begin{center}
\begin{tikzpicture}[scale=1]
    \draw[thick] (0,0) coordinate (Q) -- (4.33,0) coordinate (R) -- (0,2.5) coordinate (P) -- cycle;
    \draw (0,0) rectangle (0.25,0.25);
    \node[below left] at (Q) {$Q$};
    \node[right] at (R) {$R$};
    \node[above] at (P) {$P$};
    \node[above right] at (2.16,1.25) {$26$};
    \node[left] at (4.1,0.25) {\tiny $30^\circ$};
\end{tikzpicture}
\end{center}

\begin{solution}
This is a $30^\circ-60^\circ-90^\circ$ special right triangle. In this specific triangle type, the sides follow a fixed ratio: the side opposite the $30^\circ$ angle is $x$, the side opposite the $60^\circ$ angle is $x\sqrt{3}$, and the hypotenuse is $2x$.
\vspace{5pt}
\begin{center}
The hypotenuse $\overline{PR} = 26$.
\vspace{5pt}
$2x = 26 \implies x = 13$
\end{center}
\vspace{5pt}
The side $\overline{PQ}$ is opposite the $30^\circ$ angle, which corresponds to $x$.
\vspace{5pt}
Therefore, $PQ = 13$.
\vspace{5pt}
Answer: \colorbox{yellow!100}{13}
\end{solution}
