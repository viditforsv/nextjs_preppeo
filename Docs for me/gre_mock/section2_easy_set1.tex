%Q: 1
\question
\textbf{Quantity A: } $2^3 \times 3^2$ \\
\vspace{5pt}
\textbf{Quantity B: } $3^3 \times 2^2$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
To compare these quantities, we simply perform the arithmetic calculations for each.
\vspace{5pt}
\begin{center}
Quantity A: $2^3 \times 3^2 = 8 \times 9 = 72$
\vspace{5pt}
Quantity B: $3^3 \times 2^2 = 27 \times 4 = 108$
\end{center}
\vspace{5pt}
Since 108 is greater than 72, Quantity B is greater.

Answer: \colorbox{yellow!100}{B}
\end{solution}

\vspace{1cm}

%Q: 2
\question
$q$ is an odd integer.
\vspace{5pt}
\textbf{Quantity A: } $(-1)^q$ \\
\vspace{5pt}
\textbf{Quantity B: } $(-1)^q + 1$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
We evaluate the expressions based on the properties of negative numbers with odd exponents.
\vspace{5pt}
\begin{center}
Since $q$ is odd, $(-1)^{\text{odd}} = -1$.
\vspace{5pt}
Quantity A $= -1$
\vspace{5pt}
Quantity B $= -1 + 1 = 0$
\end{center}
\vspace{5pt}
Since 0 is greater than -1, Quantity B is greater.

Answer: \colorbox{yellow!100}{B}
\end{solution}

\vspace{1cm}

%Q: 3
\question
The circumference of a circle is $16\pi$.
\vspace{5pt}\\
\textbf{Quantity A: } The area of the circle \\
\vspace{5pt}
\textbf{Quantity B: } $60\pi$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
First, we find the radius ($r$) using the circumference formula $C = 2\pi r$.
\vspace{5pt}
\begin{center}
$16\pi = 2\pi r \implies r = 8$
\end{center}
\vspace{5pt}
Now, calculate the area using $A = \pi r^2$:
\vspace{5pt}
\begin{center}
Area $= \pi \times (8)^2 = 64\pi$
\end{center}
\vspace{5pt}
Quantity A is $64\pi$ and Quantity B is $60\pi$. Quantity A is greater.

Answer: \colorbox{yellow!100}{A}
\end{solution}

\vspace{1cm}

%Q: 4
\question
$xz = 12$ and $yz = 24$, where $z$ is a non-zero integer.
\vspace{5pt}\\
\textbf{Quantity A: } $x$ \\
\vspace{5pt}
\textbf{Quantity B: } $y$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
The relationship depends on the sign of $z$.
\vspace{5pt}
\begin{center}
If $z = 1$, then $x = 12$ and $y = 24$. (B is greater)
\vspace{5pt}
If $z = -1$, then $x = -12$ and $y = -24$. (A is greater)
\end{center}
\vspace{5pt}
Because the comparison flips depending on whether $z$ is positive or negative, the relationship cannot be determined.

Answer: \colorbox{yellow!100}{D}
\end{solution}

\vspace{1cm}

%Q: 5
\question
\begin{center}
\begin{tikzpicture}
\draw[thick] (0,0) -- (4,0) -- (1,2.5) -- cycle;
\node at (0.5, 0.2) {$x^\circ$};
\node at (3.5, 0.2) {$y^\circ$};
\node[above] at (1,2.5) {$z^\circ$};
\end{tikzpicture}
\end{center}
In the triangle above, the value of $x + y$ is 115.
\vspace{5pt}\\
\textbf{Quantity A: } $z$ \\
\vspace{5pt}
\textbf{Quantity B: } 65

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
The sum of interior angles in a triangle is always $180^\circ$.
\vspace{5pt}
\begin{center}
$x + y + z = 180$
\vspace{5pt}
$115 + z = 180$
\vspace{5pt}
$z = 180 - 115 = 65$
\end{center}
\vspace{5pt}
Quantity A is 65 and Quantity B is 65. They are equal.

Answer: \colorbox{yellow!100}{C}
\end{solution}

\question
A store owner marked down an item by 20\% and the new price is \$160. What was the original price?

\begin{tasks}(1)
\task[A)] \$180 \task[B)] \$192 \task[C)] \$200 \task[D)] \$220 \task[E)] \$240
\end{tasks}

\begin{solution}
Let $P$ be the original price. A 20\% discount means the current price is 80\% of the original.
\vspace{5pt}
\begin{center}
$0.80P = 160$
\vspace{5pt}
$P = \dfrac{160}{0.8} = \dfrac{1600}{8} = 200$
\end{center}

Answer: \colorbox{yellow!100}{C}
\end{solution}

\question
A machine produces 20 cans per hour. Each box fits 10 cans. If the machine ran for 2 hours, how many boxes are needed to pack all the cans?

\begin{tasks}(1)
\task[A)] 2 \task[B)] 4 \task[C)] 10 \task[D)] 20 \task[E)] 40
\end{tasks}

\begin{solution}
Calculate total production first, then divide by box capacity.
\vspace{5pt}
\begin{center}
Total Cans $= 20 \text{ cans/hr} \times 2 \text{ hours} = 40 \text{ cans}$
\vspace{5pt}
Total Boxes $= \dfrac{40 \text{ cans}}{10 \text{ cans/box}} = 4 \text{ boxes}$
\end{center}

Answer: \colorbox{yellow!100}{B}
\end{solution}

%Q: 9
\question
What is the area of a regular hexagon with side length 4?

\begin{tasks}(1)
\task[A)] $12\sqrt{3}$ \task[B)] $16\sqrt{3}$ \task[C)] $24\sqrt{3}$ \task[D)] $32\sqrt{3}$ \task[E)] $48\sqrt{3}$
\end{tasks}

\begin{solution}
Use the area formula for a regular hexagon: $\text{Area} = \dfrac{3\sqrt{3}}{2}s^2$.
\vspace{5pt}
\begin{center}
$\text{Area} = \dfrac{3\sqrt{3}}{2} \times (4)^2$
\vspace{5pt}
$\text{Area} = \dfrac{3\sqrt{3}}{2} \times 16 = 24\sqrt{3}$
\end{center}

Answer: \colorbox{yellow!100}{C}
\end{solution}

Indicate \textbf{all} such expressions.

\task[A)] $ab$
\task[B)] $a + b$
\task[C)] $a^b$
\end{tasks}

\vspace{5pt}
\begin{center}
Statement A: Any number multiplied by an even number is even. (True)
\vspace{5pt}
Statement B: If $a=3$ (odd) and $b=2$, sum is 5. (False)
\vspace{5pt}
Statement C: If $a=3$ and $b=2$, $3^2 = 9$. (False)
\end{center}
Only $ab$ must be even.

Answer: \colorbox{yellow!100}{A}
\end{solution}

\question
What is the area of a circle that has a circumference of $12\pi$?

\begin{solution}
\begin{center}
$12\pi = 2\pi r \implies r = 6$
\vspace{5pt}
Area $= \pi r^2 = \pi \times 6^2 = 36\pi$
\end{center}

Answer: \colorbox{yellow!100}{$36\pi$}
\end{solution}

\begin{solution}
\begin{center}
Current Total Marketing $= 25 \times 60,000 = 1,500,000$

New Average $= 1,800,000 / 30 = 60,000$
\end{center}
Since the new salaries match the existing average, the average stays the same.
