%Q: 1
\question
A car travels at a constant speed of 40 miles per hour.
\vspace{5pt}
\textbf{Quantity A: } The time it takes to travel 10 miles \\
\vspace{5pt}
\textbf{Quantity B: } 15 minutes

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
First, calculate the time for Quantity A using the formula $\text{Time} = \text{Distance} / \text{Speed}$.
\vspace{5pt}
\begin{center}
$\text{Time} = 10 / 40 = 0.25 \text{ hours}$
\vspace{5pt}
Convert hours to minutes: $0.25 \times 60 = 15 \text{ minutes}$
\end{center}
\vspace{5pt}
Quantity A is 15 minutes and Quantity B is 15 minutes. Both quantities are equal.

Answer: \colorbox{yellow!100}{C}
\end{solution}

\vspace{1cm}

%Q: 2
\question
\textbf{Quantity A: } The average (arithmetic mean) of 10, 20, 30, and 40 \\
\vspace{5pt}
\textbf{Quantity B: } 25

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
Calculate the average for Quantity A by summing the numbers and dividing by the count.
\vspace{5pt}
\begin{center}
$\text{Sum} = 10 + 20 + 30 + 40 = 100$
\vspace{5pt}
$\text{Average} = 100 / 4 = 25$
\end{center}
\vspace{5pt}
Since Quantity A is 25 and Quantity B is 25, they are equal.

Answer: \colorbox{yellow!100}{C}
\end{solution}

\vspace{1cm}

%Q: 3
\question
\begin{center}
\begin{tikzpicture}
    \draw[thick] (0,0) circle (1.5);
    \draw[thick] (0,0) -- (1.5,0) node[midway, below] {$r$};
    \node at (0,-2) {Area $= 25\pi$};
\end{tikzpicture}
\end{center}
\vspace{5pt}
\textbf{Quantity A: } The circumference of the circle \\
\vspace{5pt}
\textbf{Quantity B: } $10\pi$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
First, find the radius from the given area formula $A = \pi r^2$.
\vspace{5pt}
\begin{center}
$25\pi = \pi r^2 \implies r^2 = 25 \implies r = 5$
\vspace{5pt}
Now, find the circumference using $C = 2\pi r$:
\vspace{5pt}
$C = 2 \times \pi \times 5 = 10\pi$
\end{center}
\vspace{5pt}
Both quantities are equal to $10\pi$.

Answer: \colorbox{yellow!100}{C}
\end{solution}

%Q: 4
\question
$x$ and $y$ are positive integers such that $x + y = 10$.
\vspace{5pt}
\textbf{Quantity A: } $xy$ \\
\vspace{5pt}
\textbf{Quantity B: } 25

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
Test different integer values for $x$ and $y$ that sum to 10.
\vspace{5pt}
\begin{center}
If $x=5, y=5$, then $xy = 25$ (Equal to B)
\vspace{5pt}
If $x=4, y=6$, then $xy = 24$ (B is greater)
\vspace{5pt}
If $x=1, y=9$, then $xy = 9$ (B is greater)
\end{center}
\vspace{5pt}
The product $xy$ is at most 25, but can be less. Therefore, Quantity B is greater than or equal to Quantity A, making the relationship not fixed to a single comparison. Wait, Choice B is 25. $xy$ can be 25, but it can also be 9. Since B is sometimes equal and sometimes greater, the relationship cannot be determined.

Answer: \colorbox{yellow!100}{D}
\end{solution}

%Q: 5
\question
\textbf{Quantity A: } $3^4$ \\
\vspace{5pt}
\textbf{Quantity B: } $4^3$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
Calculate each exponent value.
\vspace{5pt}
\begin{center}
$3^4 = 3 \times 3 \times 3 \times 3 = 81$
\vspace{5pt}
$4^3 = 4 \times 4 \times 4 = 64$
\end{center}
\vspace{5pt}
Since 81 is greater than 64, Quantity A is greater.

Answer: \colorbox{yellow!100}{A}
\end{solution}

%Q: 6
\question
What is the units digit of $2^{50}$?

\begin{tasks}(1)
\task[A)] 2 \task[B)] 4 \task[C)] 6 \task[D)] 8 \task[E)] 0
\end{tasks}

\begin{solution}
Analyze the cycle of units digits for powers of 2.
\vspace{5pt}
\begin{center}
$2^1 = 2, 2^2 = 4, 2^3 = 8, 2^4 = 16 \to 6$
\vspace{5pt}
The cycle is 2, 4, 8, 6 (length 4).
\vspace{5pt}
Divide exponent 50 by 4: $50 / 4 = 12$ with remainder 2.
\vspace{5pt}
The 2nd digit in the cycle is 4.
\end{center}

Answer: \colorbox{yellow!100}{B}
\end{solution}
