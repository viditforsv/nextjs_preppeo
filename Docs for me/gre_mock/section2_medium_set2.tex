%Q: 1
\question
\[ |x|y > x|y| \]
\vspace{5pt}
\textbf{Quantity A: } $(x + y)^2$ \\
\vspace{5pt}
\textbf{Quantity B: } $(x - y)^2$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
First, analyze the given inequality $|x|y > x|y|$. 
\vspace{5pt}
\begin{center}
For $|x|y$ to be strictly greater than $x|y|$, $y$ must be positive. 
\vspace{5pt}
If $y$ is positive, we can divide both sides by $y$ to get $|x| > x$. 
\vspace{5pt}
This condition $|x| > x$ is only true if $x$ is a negative number. 
\vspace{5pt}
Therefore, we have established that $x < 0$ and $y > 0$.
\end{center}
\vspace{5pt}
Now, expand the two quantities:
\vspace{5pt}
\begin{center}
Quantity A: $(x + y)^2 = x^2 + 2xy + y^2$ \\
\vspace{5pt}
Quantity B: $(x - y)^2 = x^2 - 2xy + y^2$
\end{center}
\vspace{5pt}
Compare $2xy$ and $-2xy$. Since $x$ is negative and $y$ is positive, their product $xy$ is negative.
\vspace{5pt}
This makes $2xy$ negative and $-2xy$ positive. Any positive number is greater than a negative number.

Answer: \colorbox{yellow!100}{B}
\end{solution}

%Q: 2
\question
A dataset of tree heights is normally distributed. One standard deviation below the mean is 15 feet, and one standard deviation above the mean is 21 feet.
\vspace{5pt}
\textbf{Quantity A: } The median height of the trees \\
\vspace{5pt}
\textbf{Quantity B: } 18 feet

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
In a perfectly normal distribution, the Mean and the Median are identical.
\vspace{5pt}
\begin{center}
The mean ($\mu$) is exactly halfway between $(\mu - \sigma)$ and $(\mu + \sigma)$.
\vspace{5pt}
$\text{Mean} = \dfrac{15 + 21}{2} = \dfrac{36}{2} = 18$
\end{center}
\vspace{5pt}
Since Median = Mean in this distribution, Quantity A is 18.

Answer: \colorbox{yellow!100}{C}
\end{solution}

\question
\begin{center}
\begin{tikzpicture}[scale=1.5]
    \draw[thick] (0,0) circle (1);
    \draw[thick] (0,0) -- (1,0) node[midway, below] {$r$};
    \draw[thick] (0,0) -- (0.5, 0.866);
    \draw (0.3,0) arc (0:60:0.3);
    \node at (0.45, 0.25) {$60^\circ$};
\end{tikzpicture}
\end{center}
In the circle above, the length of the minor arc shown is $4\pi$.
\vspace{5pt}
\textbf{Quantity A: } The circumference of the circle \\
\vspace{5pt}
\textbf{Quantity B: } $24\pi$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
The ratio of an arc length to the total circumference is proportional to the ratio of the central angle to $360^\circ$.
\vspace{5pt}
\begin{center}
$\dfrac{\text{Arc Length}}{\text{Circumference}} = \dfrac{\text{Angle}}{360}$
\vspace{5pt}
$\dfrac{4\pi}{C} = \dfrac{60}{360} = \dfrac{1}{6}$
\vspace{5pt}
$C = 6 \times 4\pi = 24\pi$
\end{center}
\vspace{5pt}
Quantity A is $24\pi$ and Quantity B is $24\pi$.

Answer: \colorbox{yellow!100}{C}
\end{solution}

%Q: 4
\question
$n$ is a positive integer.
\vspace{5pt}
\textbf{Quantity A: } $(-1)^{4n} \times (-1)^{202}$ \\
\vspace{5pt}
\textbf{Quantity B: } $(3)^3 \times (-5)^5$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
Evaluate each quantity by considering the signs of the results.
\vspace{5pt}
\begin{center}
Quantity A: $4n$ is always even, and 202 is even. $(-1)^{\text{even}} = 1$.
\vspace{5pt}
$1 \times 1 = 1$ (Positive)
\vspace{10pt}
Quantity B: $3^3 = 27$ (Positive). $(-5)^5$ is a negative number raised to an odd power, which is negative.
\vspace{5pt}
$\text{Positive} \times \text{Negative} = \text{Negative}$
\end{center}
\vspace{5pt}
Since 1 is greater than any negative number, Quantity A is greater.

Answer: \colorbox{yellow!100}{A}
\end{solution}

\question
$xz = 24$ and $yz = 36$, where $z \neq 0$.
\vspace{5pt}
\textbf{Quantity A: } $x/y$ \\
\vspace{5pt}
\textbf{Quantity B: } $2/3$

\begin{tasks}(1)
\task[A)] Quantity A is greater
\task[B)] Quantity B is greater
\task[C)] The two quantities are equal
\task[D)] The relationship cannot be determined
\end{tasks}

\begin{solution}
Divide the first equation by the second equation to isolate the ratio of $x$ to $y$.
\vspace{5pt}
\begin{center}
$\dfrac{xz}{yz} = \dfrac{24}{36}$
\vspace{5pt}
Since $z$ is non-zero, it cancels out:
\vspace{5pt}
$\dfrac{x}{y} = \dfrac{24}{36}$
\vspace{5pt}
Simplify the fraction by dividing by 12:
\vspace{5pt}
$\dfrac{x}{y} = \dfrac{2}{3}$
\end{center}
\vspace{5pt}
Quantity A is equal to Quantity B.

Answer: \colorbox{yellow!100}{C}
\end{solution}

%Q: 6
\question
A contractor has two teams. Team A can pave a road in 12 days, and Team B can pave it in 18 days. If Team A works alone for 2 days and then Team B joins to finish, how many total days does the project take?

\begin{tasks}(1)
\task[A)] 6 \task[B)] 8 \task[C)] 10 \task[D)] 12 \task[E)] 14
\end{tasks}

\begin{solution}
\begin{center}
Team A rate $= 1/12$. Team B rate $= 1/18$.
\vspace{5pt}
Work done by A in 2 days $= 2 \times (1/12) = 1/6$.
\vspace{5pt}
Remaining work $= 1 - 1/6 = 5/6$.
\vspace{5pt}
Combined rate $= 1/12 + 1/18 = 3/36 + 2/36 = 5/36$.
\vspace{5pt}
Time for combined work $= (5/6) / (5/36) = (5/6) \times (36/5) = 6 \text{ days}$.
\vspace{5pt}
Total time $= 2 + 6 = 8 \text{ days}$.
\end{center}

Answer: \colorbox{yellow!100}{B}
\end{solution}

If $f(x) = x^2 + 5$, what is the value of $f(a+b) - f(a-b)$?

\begin{tasks}(1)
\task[A)] $2b^2 + 10$ \task[B)] $4ab$ \task[C)] $4ab + 10$ \task[D)] $2a^2 + 2b^2$ \task[E)] $0$
\end{tasks}

\begin{solution}
\begin{center}
$f(a+b) = (a+b)^2 + 5 = a^2 + 2ab + b^2 + 5$
\vspace{5pt}
$f(a-b) = (a-b)^2 + 5 = a^2 - 2ab + b^2 + 5$
\vspace{5pt}
$(a^2 + 2ab + b^2 + 5) - (a^2 - 2ab + b^2 + 5) = 4ab$
\end{center}

Answer: \colorbox{yellow!100}{B}
\end{solution}

\end{center}

What is the remainder when $4^{20}$ is divided by 10?

\begin{tasks}(1)
\task[A)] 0 \task[B)] 2 \task[C)] 4 \task[D)] 6 \task[E)] 8
\end{tasks}

\begin{solution}
Dividing by 10 is equivalent to finding the units digit.
\vspace{5pt}
\begin{center}
$4^1 = 4, 4^2 = 16 \to 6, 4^3 = 64 \to 4, 4^4 = 256 \to 6$
\vspace{5pt}
The cycle is 4 (odd exponent) and 6 (even exponent).
\vspace{5pt}
Since 20 is even, the units digit is 6.
\end{center}

Answer: \colorbox{yellow!100}{D}
\end{solution}

Indicate \textbf{all} such expressions.

B: If $x=2$, $x^2/3 = 4/3$, not necessarily an integer. (False)
\vspace{5pt}
C: $3g - 25g^2$. Since $g$ is integer, this is an integer. (True)
\end{center}

Answer: \colorbox{yellow!100}{A, C}
