% Question 1
% Category: Exam Level | Difficulty: Medium | Question Type: Numeric Entry
\question
\begin{center}
\begin{tikzpicture}[scale=0.8]
\draw[thick] (0,0) node[left]{$A$} -- (3,0) node[below]{$C$} -- (3,5) node[above]{$B$} -- cycle;
\draw (3,0) rectangle (2.7,0.3);
\node at (1.5,-0.4) {3 units};
\node at (3.8,2.5) {?};
\node at (0.8,2.8) {$\sqrt{34}$ units};
\end{tikzpicture}
\end{center}
In the figure shown, $AB = \sqrt{34}$ units and $AC = 3$ units. If triangle $ABC$ is a right triangle with the right angle at $C$, what is the length of segment $BC$?

\begin{solution}
\textbf{Conceptual Explanation:} \\
The Pythagorean Theorem states that for any right triangle with legs $a$ and $b$ and hypotenuse $c$, the relationship $a^2 + b^2 = c^2$ must hold. In this problem, we are given the hypotenuse ($AB$) and one leg ($AC$), and we need to solve for the missing leg ($BC$).

\vspace{30pt}

\textbf{Calculation and Logic:} \\
Identify the known values: $c = \sqrt{34}$ and $a = 3$. Set up the equation: $3^2 + (BC)^2 = (\sqrt{34})^2$. Calculate the squares: $9 + (BC)^2 = 34$. Isolate the missing term by subtracting 9 from both sides: $(BC)^2 = 34 - 9$. This gives $(BC)^2 = 25$. To find $BC$, take the square root of both sides: $BC = \sqrt{25} = 5$. The length of segment $BC$ is 5 units.

\vspace{30pt}

Answer: \colorbox{yellow!100}{5}
\end{solution}

\end{solution}

A coordinate plane has points at $(0,0)$ and $(8,15)$. What is the distance between these two points?

\begin{solution}
\textbf{Conceptual Explanation:} \\
The distance formula $d = \sqrt{(x_2-x_1)^2 + (y_2-y_1)^2}$ is derived directly from the Pythagorean Theorem, where the difference in $x$-coordinates is one leg and the difference in $y$-coordinates is the other.

\vspace{30pt}

\textbf{Calculation and Logic:} \\
Leg 1 (horizontal distance): $8 - 0 = 8$. Leg 2 (vertical distance): $15 - 0 = 15$. Apply the theorem: $8^2 + 15^2 = c^2 \Rightarrow 64 + 225 = c^2 \Rightarrow 289 = c^2$. The distance $c = \sqrt{289} = 17$.

\vspace{30pt}

Answer: \colorbox{yellow!100}{17}
\end{solution}

\question
\begin{center}
\begin{tikzpicture}[scale=0.8]
\draw[thick] (0,0) -- (4,0) node[midway, below]{32$^\circ$} -- (4,2.5) -- cycle;
\draw[thick] (6,2.5) -- (6,0) node[midway, below]{58$^\circ$} -- (8,0) -- cycle;
\node at (0.2,1.5) {$A$}; \node at (7.8,1.5) {$E$};
\node at (4.3,2.5) {$B$}; \node at (5.7,2.5) {$T$};

Triangles $ABC$ and $DEF$ are right triangles. Which of the following is equal to the ratio $\frac{BC}{AB}$?
\begin{tasks}(2)
    \task $\frac{DE}{DF}$
    \task $\frac{DF}{DE}$
    \task $\frac{DF}{EF}$
    \task $\frac{EF}{DE}$
\end{tasks}

\begin{solution}
\textbf{Conceptual Explanation:} \\
In right triangles, the ratio of sides is determined by the acute angles. If two right triangles have an identical acute angle, they are similar, and their side ratios (trigonometric functions) are equal. Note that $32^\circ + 58^\circ = 90^\circ$, meaning these triangles are complementary.

\vspace{30pt}

\textbf{Calculation and Logic:} \\
In $\triangle ABC$, $\frac{BC}{AB}$ is $\sin(32^\circ)$ or $\cos(58^\circ)$. In $\triangle DEF$, $\cos(58^\circ)$ is the ratio of adjacent leg to hypotenuse, which is $\frac{DF}{DE}$. Therefore, $\frac{BC}{AB} = \frac{DF}{DE}$.

\vspace{30pt}

Answer: \colorbox{yellow!100}{B}
\end{solution}
